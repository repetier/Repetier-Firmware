462  "adc %C0,r1 \n\t"

635  static inline void spiInit(uint8_t spiRate)
636  {
637  uint8_t r = 0;
638  for (uint8_t b = 2; spiRate > b && r < 6; b <<= 1, r++);
639 
640  SET_OUTPUT(SS);
641  WRITE(SS,HIGH);
642  SET_OUTPUT(SCK);
643  SET_OUTPUT(MOSI_PIN);
644  SET_INPUT(MISO_PIN);
645 #ifdef PRR
646  PRR &= ~(1<<PRSPI);
647 #elif defined PRR0
648  PRR0 &= ~(1<<PRSPI);
649 #endif
650  // See avr processor documentation
651  SPCR = (1 << SPE) | (1 << MSTR) | (r >> 1);
652  SPSR = (r & 1 || r == 6 ? 0 : 1) << SPI2X;
