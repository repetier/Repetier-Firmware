421  inline void updateAdvanceSteps(speed_t v, uint8_t max_loops, bool accelerate) {
422 #if USE_ADVANCE
423  if(!Printer::isAdvanceActivated()) return;
424 #if ENABLE_QUADRATIC_ADVANCE
425  long advanceTarget = Printer::advanceExecuted;
426  if(accelerate) {
427  for(uint8_t loop = 0; loop < max_loops; loop++) advanceTarget += advanceRate;
428  if(advanceTarget > advanceFull)
429  advanceTarget = advanceFull;
430  } else {
431  for(uint8_t loop = 0; loop < max_loops; loop++) advanceTarget -= advanceRate;
432  if(advanceTarget < advanceEnd)
433  advanceTarget = advanceEnd;
434  }
435  long h = HAL::mulu16xu16to32(v, advanceL);
436  int tred = ((advanceTarget + h) >> 16);
437  HAL::forbidInterrupts();
438  Printer::extruderStepsNeeded += tred - Printer::advanceStepsSet;
439  if(tred > 0 && Printer::advanceStepsSet <= 0)
440  Printer::extruderStepsNeeded += Extruder::current->advanceBacklash;
441  else if(tred < 0 && Printer::advanceStepsSet >= 0)
442  Printer::extruderStepsNeeded -= Extruder::current->advanceBacklash;
443  Printer::advanceStepsSet = tred;
444  HAL::allowInterrupts();
445  Printer::advanceExecuted = advanceTarget;
446 #else
447  int tred = HAL::mulu6xu16shift16(v, advanceL);
448  HAL::forbidInterrupts();
449  Printer::extruderStepsNeeded += tred - Printer::advanceStepsSet;
450  if(tred > 0 && Printer::advanceStepsSet <= 0)
451  Printer::extruderStepsNeeded += (Extruder::current->advanceBacklash << 1);
452  else if(tred < 0 && Printer::advanceStepsSet >= 0)
453  Printer::extruderStepsNeeded -= (Extruder::current->advanceBacklash << 1);
454  Printer::advanceStepsSet = tred;
455  HAL::allowInterrupts();
456 #endif
457 #endif
458  }
